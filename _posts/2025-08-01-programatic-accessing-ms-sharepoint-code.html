class MSFileSync():
    
    def __init__(self):
        self.credentials = ClientSecretCredential(
                                tenant_id,
                                client_id,
                                client_secret,
                            )
        self.scopes = ['https://graph.microsoft.com/.default']
        self.client = GraphServiceClient(credentials=self.credentials, scopes=self.scopes)
        self.drive_id = None
        self.root_id = None
    async def get_drive_id(self) -> str:
        drive = await self.client.sites.by_site_id(sharepoint_site_id).drive.get()
        self.drive_id = drive.id  
        return self.drive_id

    async def get_root_id(self) -> str:
        root_id = await self.client.drives.by_drive_id(self.drive_id).root.get()
        self.root_id = root_id.id
        return self.root_id

    async def get_file(self, sharepoint_file: SharepointFile):
        file_stream = await self.client.drives.by_drive_id(self.drive_id).items.by_drive_item_id(sharepoint_file.file_id).content.get()
        return file_stream

While there are more pages, keep retrieving the nextLink, until none found. This could ensure that we fetch a full list.

async def get_drive_items(self, item_id) -> list:
        all_items = []
        response = await self.client.drives.by_drive_id(self.drive_id).items.by_drive_item_id(item_id).children.get()
        all_items.extend(response.value)
        # Follow nextLink if present
        # While there are more pages, keep retrieving the nextLink
        # help doc: https://github.com/microsoftgraph/msgraph-sdk-python/blob/58f61a0384bf8070a3cd51a82b23eb77c31bc21d/README.md?plain=1#L191
        while response is not None and response.odata_next_link is not None:
            # Follow the nextLink to get the next page
            response = await self.client.drives.by_drive_id(self.drive_id).items.by_drive_item_id(item_id).children.with_url(response.odata_next_link).get()
            all_items.extend(response.value)
        return all_items
