Java complains the error

"javax.net.ssl.SSLHandshakeException: no subject alternatice names matching IP address X.X.X.X found"

Add ip address into certificate's subject alternative name is a elegant way to resolve this error.

Command Line way

openssl ecparam -genkey -name secp256r1 -out $SERVER_KEY

openssl req -new -sha256 -key $SERVER_KEY -subj "/C=NL/ST=NB/O=iotcolon B.V./CN=$SERVER" -addext "subjectAltName=DNS:$SERVER,DNS:iotcolon.com,IP:192.168.1.1" -out $SERVER_CSR -config <(cat /etc/ssl/openssl.cnf <(printf "[SAN]\nsubjectAltName=DNS:$SERVER,DNS:iotcolon.com")) -reqexts SAN

openssl x509 -req -in $SERVER_CSR -CA $CA_CERT_FILE -CAkey $CA_KEY_FILE -CAcreateserial -out $SERVER_CERT -days 70000 -sha256 -extfile <(cat /etc/ssl/openssl.cnf <(printf "[SAN]\nsubjectAltName=DNS:$SERVER,DNS:iotcolon.com,IP:192.168.1.1")) -extensions SAN

openssl config way:

req_extensions = v3_req
distinguished_name = req_distinguished_name
default_bits = 4096
default_md = sha256
prompt = no
[req_distinguished_name]
C = ...
ST = ...
L = ...
OU = ...
O = ...
CN = 192.168.0.0
[ v3_req ]
basicConstraints = CA:FALSE
keyUsage = nonRepudiation, digitalSignature, keyEncipherment
extendedKeyUsage=serverAuth,clientAuth
subjectAltName = @alt_names
[alt_names]
IP.1 = 192.168.0.1
IP.2 = 192.168.0.2
IP.3 = 192.168.0.3
...
IP.254 = 192.168.0.254
